#### ❌ 问题2：无输入验证

**位置：** `src/app/api/customer-analysis/route.ts`

```typescript
// ❌ 问题：直接解析请求体，无验证
const body: CustomerAnalysisRequest = await request.json();

// ❌ 问题：仅检查空值，无类型/长度验证

  return NextResponse.json(
    { success: false, error: '请提供公司名称' },

  );

```

**风险：**
- SQL注入风险（虽然没有直接SQL，但数据可能流入数据库）
- XSS风险（数据未清洗直接返回）
- DoS风险（无请求体大小限制）

**修复建议：**
```typescript
import { z } from 'zod';

const schema = z.object({
  companyName: z.string().min(1).max(200).trim(),
  includeCompetitors: z.boolean().optional().default(false),
  includeNews: z.boolean().optional().default(false),
});

export async function POST(request: NextRequest) {
  const body = await request.json();
  const result = schema.safeParse(body);
  
  if (!result.success) {
    return NextResponse.json(
      { success: false, error: '参数错误', details: result.error.format() },
      { status: 400 }
    );
  }
  
  const { companyName, includeCompetitors, includeNews } = result.data;
  // ...
}
```

#### ❌ 问题5：API调用无超时和重试

**位置：** `src/lib/api/integration.ts`

```typescript
// ❌ 问题：fetch 没有超时控制
async function apiCall<T>(
  baseUrl: string,
  endpoint: string,
  options: RequestInit = {}
): Promise<ZAEPResponse<T>> {
  const response = await fetch(url, {  // 可能永远等待
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...options.headers,
    }
  });
}
```

**修复建议：**
```typescript
async function apiCall<T>(
  baseUrl: string,
  endpoint: string,
  options: RequestInit = {},
  timeoutMs = 10000,  // 默认10秒超时
  retries = 3         // 默认重试3次
): Promise<ZAEPResponse<T>> {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);
  
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, {
        ...options,
        signal: controller.signal,
        headers: {
          'Content-Type': 'application/json',
          ...options.headers,
        }
      });
      clearTimeout(timeoutId);
      return createResponse<T>(response.ok, await response.json());
    } catch (error) {
      if (i === retries - 1) throw error;
      await new Promise(r => setTimeout(r, 1000 * (i + 1)));  // 指数退避
    }
  }
  
  return createResponse<T>(false, undefined, {
    code: 'TIMEOUT',
    message: '请求超时'
  });
}
```

#### ⚠️ 问题7：单例模式实现不当

**位置：** `src/lib/db/connection.ts`

```typescript
// ❌ 问题：单例模式在服务端可能有问题
export class DatabasePool {
  private static instance: DatabasePool;
  private supabase = createSupabaseClient();  // 在 import 时就创建
  
  private constructor() {}
  
  static getInstance(): DatabasePool {
    if (!DatabasePool.instance) {
      DatabasePool.instance = new DatabasePool();
    }
    return DatabasePool.instance;
  }
}

// ❌ 问题：直接导出实例
export default DatabasePool.getInstance();
```

**问题：**
- Next.js 服务端渲染时，单例可能跨请求共享
- 可能导致数据泄露或连接池耗尽

**修复建议：**
```typescript
// ✅ 使用函数返回新实例
export function createDatabasePool() {
  return new DatabasePool();
}

// ✅ 或使用 React Context/Provider 模式
// app/providers.tsx
export function DatabaseProvider({ children }) {
  const [db] = useState(() => createDatabasePool());
  return (
    <DatabaseContext.Provider value={db}>
      {children}
    </DatabaseContext.Provider>
  );
}
```
